PoW_liberate_prisoners = {
	every_in_list = {
		variable = PoW
		release_from_prison = yes
		if = {
			limit = { is_landed = yes }
			capital_province = { save_scope_as = home }
		}
		else = { court_owner.capital_province = { save_scope_as = home }}
		if = {
			limit = { exists = scope:home }
			set_location = prev.var:location
			start_travel_plan = {
				destination = scope:home
				players_use_planner = no
				return_trip = no
			}
		}
	}
}
