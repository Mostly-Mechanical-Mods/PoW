PoW_liberate_prisoners = {
	if = {
		limit = { PoW_debug = yes }
		debug_log = "PoW_liberate_prisoners"
	}
	every_in_list = {
		variable = PoW
		if = { limit = { PoW_debug = yes } debug_log_scopes = no }
		if = {
			limit = { is_imprisoned = no }
			prev ?= {
				remove_list_variable = {
					name = PoW
					target = prev
				}
			}
			remove_variable = PoW
		}
		else = { escape_from_prison_effect = yes }
	}
	clear_variable_list = PoW
}
